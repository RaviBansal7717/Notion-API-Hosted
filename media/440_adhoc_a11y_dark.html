import { Component, ElementRef, Inject, OnInit, ViewChild } from '@angular/core';
import { FormControl } from '@angular/forms';
import { MatAutocompleteSelectedEvent } from '@angular/material/autocomplete';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { MonacoEditorConstructionOptions, MonacoStandaloneCodeEditor } from "@materia-ui/ngx-monaco-editor";
import { Observable } from 'rxjs';
import { filter, map, startWith } from 'rxjs/operators';
import { CodeBlock, CodeDialogData } from '../separate-entities/separate-entities.component';
import { ProgLanguagesService } from '../services/prog-languages.service';

export interface ProgLanguage {
    viewValue: string;
    monacoValue: string;
    prismValue: string;
}

@Component({
    selector: 'app-code-input',
    templateUrl: './code-input.component.html',
    styleUrls: ['./code-input.component.css']
})
export class CodeInputComponent implements OnInit {

    // @ViewChild('codeBlock', { static: false }) codeBlock!: ElementRef;

    monacoEditor!: MonacoStandaloneCodeEditor;
    langFormControl = new FormControl('');
    filteredLanguages!: Observable;
    languagesRepository: ProgLanguage[] = [];
    defaultLang: string = "Python"
    langNotFoundString: string = "No such language"
    selectedMonacoLang: string = "";
    selectedPrismLang: string = "";
    codeSnippet: string = "";
    // codeBlock: CodeBlock = { language: "", code: "" };
    isCodeBlockSaved: boolean = true;

    editorOptions: MonacoEditorConstructionOptions = {
    theme: "vs-dark",
    roundedSelection: true,
    automaticLayout: true,
    autoIndent: "full",
    language: this.selectedMonacoLang
    };

    constructor(public dialogRef: MatDialogRef,
    @Inject(MAT_DIALOG_DATA) public codeDialogData: CodeDialogData,
    private progLangService: ProgLanguagesService) { 
    }
    
    editorInit(editor: MonacoStandaloneCodeEditor) {
    this.monacoEditor = editor;
    }


    ngOnInit(): void {
    this.progLangService.fetchProgrammingLangData().subscribe({
        next: (response) => {
        console.log("Success : ", response);
        this.languagesRepository = response;

        // Angular Material Autocomplete logic to filter languages
        this.filteredLanguages = this.langFormControl.valueChanges.pipe(
            startWith(''),
            map(value => this._filterLanguages(value || '')),
        );
        },
        error: (error) => console.log("Error: ", error),
        complete: () => console.log("Complete")
    });
    }

    private _filterLanguages(value: string): ProgLanguage[] {
    console.log("In filter : ", value);
    let arr = this.languagesRepository.filter(
        (language) => language.viewValue.toLowerCase().indexOf(value.toLowerCase()) === 0
    );
    return arr.length ? arr : [{ viewValue: this.langNotFoundString, monacoValue: this.defaultLang, prismValue: this.defaultLang}];
    }

    langChange($event: MatAutocompleteSelectedEvent) {
    let selection = $event.option.value;
    if (selection === this.langNotFoundString) {
        [this.selectedMonacoLang, this.selectedPrismLang] = this.mapLangViewToValue(this.defaultLang);
        console.log("Default Case");
    }
    else {
        [this.selectedMonacoLang, this.selectedPrismLang] = this.mapLangViewToValue(selection);
    }

    this.editorOptions = { ...this.editorOptions, language: this.selectedMonacoLang };
    this.codeDialogData.codeBlock.language = this.selectedPrismLang;
    }

    codeEdit($event: Event) {
    console.log("\n\nHere : ");
    this.codeDialogData.codeBlock.code = this.codeSnippet;
    this.codeDialogData.codeBlock.language = this.selectedPrismLang;
    console.log($event);
    }



    mapLangViewToValue(selectedLang: string): string[] {
    let langObj = this.languagesRepository.find((language) => language.viewValue === selectedLang);
    return [langObj!.monacoValue, langObj!.prismValue];
    }

    onClose(): void {
    this.dialogRef.close(!this.isCodeBlockSaved);
    }

    // langChange function triggered on Mat Select Change (earlier logic)
    // langChange($event: MatSelectChange) {
    //   this.editorOptions.language = this.selectedLang;
    // }

}
@Configuration
@EnableWebSecurity
public class BasicConfiguration {

    @Bean
    public InMemoryUserDetailsManager userDetailsService(PasswordEncoder passwordEncoder) {
        UserDetails user = User.withUsername("user")
            .password(passwordEncoder.encode("password"))
            .roles("USER")
            .build();

        UserDetails admin = User.withUsername("admin")
            .password(passwordEncoder.encode("admin"))
            .roles("USER", "ADMIN")
            .build();

        return new InMemoryUserDetailsManager(user, admin);
    }

    @Bean
    public SecurityFilterChain filterChain(HttpSecurity http) throws Exception {
        http.authorizeRequests()
            .anyRequest()
            .authenticated()
            .and()
            .httpBasic();
        return http.build();
    }

    @Bean
    public PasswordEncoder passwordEncoder() {
        PasswordEncoder encoder = PasswordEncoderFactories.createDelegatingPasswordEncoder();
        return encoder;
    }
}

function toggleAccordionTOC(btnElement) {
    btnElement.classList.toggle("active");
    let panel = btnElement.nextElementSibling;
    if (panel.style.maxHeight) {
        panel.style.maxHeight = null;
        panel.style.borderColor = "transparent";
    } else {
        panel.style.maxHeight = panel.scrollHeight + "px";
        panel.style.borderColor = "lightgray";
    }
}

function myFunction() {
    var winScroll = document.body.scrollTop || document.documentElement.scrollTop;
    var height = document.documentElement.scrollHeight - document.documentElement.clientHeight;
    var scrolled = (winScroll / height) * 100;
    document.getElementById("progressBar").style.width = scrolled + "%";
}
